package com.app.example.idempotency.generators;

import com.app.example.idempotency.exception.ArgumentnotFoundException;
import org.springframework.http.HttpHeaders;
import org.springframework.util.StringUtils;

public class DefaultIdempotencyKeyGenerator implements IdempotencyKeyGenerator {

	private static final String IDEMPOTENCY_DEFALUT_HEADER = "X-Idempotency-Key";

	@Override
	public String generateKey(HttpHeaders headers) {
		Object idempotencyHeader = headers.get(IDEMPOTENCY_DEFALUT_HEADER);
		if (StringUtils.isEmpty(idempotencyHeader)) {
			throw new ArgumentnotFoundException(IDEMPOTENCY_DEFALUT_HEADER);
		}
		return idempotencyHeader.toString();
	}
}
